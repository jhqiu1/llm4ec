(function() {
  var btns = document.querySelectorAll('.tl-legend-btn');
  var entries = document.querySelectorAll('.tl-entry');
  var reset = document.getElementById('tl-reset');

  function updateVisibility() {
    var active = {};
    btns.forEach(function(b) {
      if (b.classList.contains('active')) active[b.dataset.filter] = true;
    });

    entries.forEach(function(e) {
      if (active[e.dataset.cat]) {
        e.classList.remove('hidden');
      } else {
        e.classList.add('hidden');
      }
    });
  }

  btns.forEach(function(b) {
    b.addEventListener('click', function() {
      b.classList.toggle('active');
      updateVisibility();
    });
  });

  reset.addEventListener('click', function() {
    btns.forEach(function(b) { b.classList.add('active'); });
    updateVisibility();
  });
})();
</script>
